def create_table():
    import sqlite3
    conn = sqlite3.connect('fiscalizacao.db')   
    c = conn.cursor()
    c.execute('CREATE TABLE IF NOT EXISTS car (id INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, proprietario TEXT, '\
              'cpf TEXT, placa TEXT, convencional_ou_autonomo TEXT)')
    c.close()
    conn.close()

    conn = sqlite3.connect('fiscalizacao.db')

              (proprietario, cpf, placa, convencional_ou_autonomo))

def Autenticacao(text):

    conn = sqlite3.connect('fiscalizacao.db')   
    c = conn.cursor()
    
    id = []
    proprietario = []
    cpf=[]
    placa = []
    classificacao = []
    c.execute("SELECT * FROM car")
  # Mostra

    for linha in c.fetchall():
        id.append(linha[0])
        proprietario.append(linha[1])
        cpf.append(linha[2])
        placa.append(linha[3])
        classificacao.append(linha[4])
      
    df_garagem = pd.DataFrame({'ID':id ,'Proprietario':proprietario, 'CPF': cpf,'Placa': placa, 'Classificacao':classificacao})

    df = df_garagem[df_garagem['Placa']==text]
    df = df.drop(columns=['ID'])
    df = df.drop_duplicates()
